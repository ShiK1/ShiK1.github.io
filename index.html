字符串hash

假设字符串是由字母A~Z和字母a~z组成，那么可以设

a~z为0~25,A~Z为26~51，这样即可将字母字符串转换成52进制的数字，然后转换成10进制，就可以实现将字符串映射为整数，因为不同字符串所组成的数字一定是不同的。转换进制后也不同

//len是字符串的长度，所以最大的字符串的数字为52^len-1。（全部为最大字母）
long hashFunc (char s[], int len){
    lont id = 0;
    for(int i = 0; i < len; i++){
        if (s[i]<='z'&&s[i]>='a')
            id = id*52 + (s[i]-'a');
        else if(s[i]>='A'&&s[i]<='Z')
            id = id*52 + (s[i]-'A'+26);
        //else  如果还有除了字母以外的字符，可以再添加进制
    }
    return id;
}

分数的表示和化简

typedef struct fraction{
    int up; //分子
    int down; // 分母
}Fraction_number,F[size],f;

给分数一些规定

  1. 如果分母down为负数，则令分子和分母都变为相反数；
  2. 如果分子为0，则令分母为1；
  3. 约分：求出分子和分母绝对值的最大公约数d，然后令分子和分母同除以d。
Fraction_number Simplification(Fraction_number a){
    if(a.down < 0){
        a.down = -a.down;
        a.up = -a.up;
    }
    if(a.up==0)
        a.down = 1;
    else{
        int d = gcd(abs(a.up), abs(a.down));
        a.up /= d;
        a.down /= d;
    }
    return a;
}

分数的四则运算

1.分数的加法

f fsum(f f1, f f2){
    f result;
    result.up = f1.up*f2.down + f2.up*f1.down;
    result.down = f1.down*f2.down;
    return Simplification(result); //化简
}

2.减法

f subtract(f f1, f f2){
    f result;
    result.up = f1.up*f2.down - f2.up*f1.down;
    result.down = f1.down*f2.down;
    return Simplification(result); //化简
}

3.乘法

f multiply(f f1, f f2){
    f result;
    result.up = f1.up * f2.up;
    result.down = f1.down * f2.down;
    return Simplification(result); //化简
}

4.除法

注意f2的分子如果为0，则直接输出错误。

f division(f f1, f f2){
    f result;
    result.up = f1.up * f2.down;
    result.down = f1.down * f2.up;
    return Simplification(result); //化简
}
